function createBalloon(data) {
  const balloon = document.createElement("div");
  balloon.classList.add("balloon");
  balloon.style.background = data.color;
  balloon.style.left = Math.random() * 90 + "vw";
  balloon.style.animationDuration = (6 + Math.random() * 5) + "s";

  balloon.addEventListener("click", () => {
    balloon.remove();
    showModal(data.image, data.message);
  });

  document.body.appendChild(balloon);
}

function showModal(img, message) {
  document.getElementById("popupImage").src = img;
  document.getElementById("popupMessage").innerText = message;
  document.getElementById("modal").style.display = "flex";
}

function closeModal() {
  document.getElementById("modal").style.display = "none";
}

setInterval(() => {
  const randomBalloon = balloonData[Math.floor(Math.random() * balloonData.length)];
  createBalloon(randomBalloon);
}, 1000);
